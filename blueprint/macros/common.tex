% --- common preamble for BOTH web + print builds ---

% If the blueprint LaTeX package is available, use it; otherwise define the
% theorem-like environments ourselves (so the PDF build always works).
\IfFileExists{blueprint.sty}{
  % Use the blueprint package's environments
  \usepackage[thms=dfn+lem+prop+thm+cor+rmk+ex]{blueprint}
}{
  % Fallback: provide dfn/thm/etc via amsthm so labels/counters work
  \usepackage{amsthm}
  \newtheorem{thm}{Theorem}[section]
  \newtheorem{lem}[thm]{Lemma}
  \newtheorem{prop}[thm]{Proposition}
  \newtheorem{cor}[thm]{Corollary}
  \theoremstyle{definition}
  \newtheorem{dfn}[thm]{Definition}
  \theoremstyle{remark}
  \newtheorem*{rmk}{Remark}
  \newtheorem{ex}[thm]{Example}
}
